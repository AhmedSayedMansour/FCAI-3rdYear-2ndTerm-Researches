#include <stdio.h>
#include <stdlib.h>
#include <omp.h>
#define THREADS 32
int **alloc(int rows, int cols)
{
    int *data = (int *)malloc(rows*cols*sizeof(int));
    int **array= (int **)malloc(rows*sizeof(int*));
    int i;
    for (i=0; i<rows; i++)
        array[i] = &(data[cols*i]);

    return array;
}
void free2D(int **mat)
{
    free(mat[0]);
    free(mat);

}
void printMat(int **mat,int row, int col)
{
    int i,j;
    for (i=0; i<row; i++)
    {
        for (j=0; j<col; j++)
        {
            printf("%d   ",mat[i][j]);
        }
        printf("\n");
    }

}

int main(int argc, char * argv[])
{

    int n,r,m,i,j,k;
    int **mat1;
    int **mat2;
    int **mat3;

    int counter;

    n=1024;
    r=1024;
    m=1024;

    mat1=alloc(n,r);
    mat2=alloc(r,m);
    mat3=alloc(n,m);

    for (i=0; i<n; i++)
    {
        for (j=0; j<r; j++)
        {
            mat1[i][j]=1;
        }
    }
    for (i=0; i<r; i++)
    {
        for (j=0; j<m; j++)
        {
            mat2[i][j]=5;
        }
    }



    double start=omp_get_wtime();
    #pragma omp parallel shared(mat1,mat2,mat3) private(i,j,k)
    {
#pragma omp for  schedule(static)
        for (i=0; i<m; i++)
        {
            for (j=0; j<n; j++)
            {
                for (k=0; k<r; k++)
                {
                    mat3[i][j]+=mat1[i][k]*mat2[k][j];
                }
            }

        }


    }

    double end=omp_get_wtime();
   printf("%f  \n",end-start);
   // printMat(mat3,m,n);
    free2D(mat1);
    free2D(mat2);
    free2D(mat3);


    return 0;
}
